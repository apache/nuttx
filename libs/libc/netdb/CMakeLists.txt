if("${CONFIG_LIBC_NETDB}" STREQUAL "y")

	set(CSRCS)

	# Add the netdb C files to the build
	list(APPEND CSRCS
		lib_netdb.c lib_gethostbyname.c lib_gethostbynamer.c
		lib_getservbyname.c lib_getservbynamer.c
		lib_gaistrerror.c lib_freeaddrinfo.c lib_getaddrinfo.c
	)

	# Add host file support
	if("${CONFIG_NETDB_HOSTFILE}" STREQUAL "y")
		list(APPEND CSRCS
			lib_gethostbyaddr.c lib_gethostbyaddrr.c lib_parsehostfile.c
		)
	endif()

	# Add DNS lookup support

	if("${CONFIG_NETDB_DNSCLIENT}" STREQUAL "y")
		list(APPEND CSRCS
			lib_dnsinit.c lib_dnsbind.c lib_dnsquery.c lib_dnsaddserver.c
			lib_dnsforeach.c lib_dnsnotify.c
		)

		if(NOT "${CONFIG_NETDB_DNSCLIENT_ENTRIES}" STREQUAL "0")
			list(APPEND CSRCS lib_dnscache.c)
		endif()
	endif()

	# Add the netdb directory to the build
	set_parent_srcs(${CSRCS})

endif()
