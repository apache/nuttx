# generate files

  OUTPUT ${SYSCALLS}
  COMMAND ${CMAKE_BINARY_DIR}/bin/mksyscall -w
          ${CMAKE_CURRENT_LIST_DIR}/../syscall.csv
  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR} DEPENDS nuttx_host_tools)
